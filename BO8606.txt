བོད་རྒྱ་གཉིས་ནི་གནའ་སྔ་མོ་ནས་གླིང་ཆེན་ཨེ་ཤེ་ཡའི་ནང་མིང་གྲགས་ཡོད་པའི་ཁྱིམ་མཚེས་རྒྱལ་ཁབ་ཐ་དད་དུ་ གྲུབ་ཡོད་ལ། རྒྱ་བོད་ས་མཚམས་སུ་དམག་འཁྲུག་ཀྱང་ཡང་ཡང་བྱུང་མྱོང་ཡོད་པ་དང་། ས་མཚམས་དབྱེ་འབྱེད་ཀྱི་ཞི་ འགྲིག་བགྱིས་པའི་གྲོས་དོན་"བོད་བོད་ཡུལ་ན་སྐྱིད། རྒྱ་རྒྱ་ཡུལ་ན་སྐྱིད།"ཅེས་བོད་རྒྱ་ཡིག་རིགས་སོ་སོར་བཀོད་དེ་ བཙུགས་པའི་རྡོ་རིང་ཞིག་དུས་ད་ལྟའང་བོད་ཀྱི་རྒྱལ་ས་ལྷ་ལྡན་གཙུག་ལག་ཁང་གི་མདུན་དུ་ཀུན་གྱིས་མཐོང་ཆོས་སུ་ གནས་ཡོད། ཡིན་ནའང་རྒྱ་ནག་དམར་གཞུང་གིས་བོད་གངས་ཅན་ཞིང་གི་སྣོད་བཅུད་ཡོངས་རྫོགས་ཧམ་ཟ་བྱ་རྒྱུའི་ངན་ སེམས་བཅངས་ཏེ་རྒྱ་ནག་དམར་གཞུང་གིས་ཕྱི་ལོ་ ༡༩༤༩ ལོར་བོད་ནང་བཙན་འཛུལ་གྱིས་ཞི་བའི་བཅིངས་འགྲོལ་ གཏང་རྒྱུ་ཟེར་བར་ཁ་གཡར་ཏེ་བོད་ཆོལ་གསུམ་ཡོངས་སུ་དྲག་པོའི་དཔུང་འཇུག་བྱས་ཏེ་བོད་མི་འབུམ་ཕྲག་མང་པོའི་ཚེ་ སྲོག་འཕྲོག་པ་མ་ཟད། ཕྱི་ལོ་ ༡༩༥༩ ཟླ་ ༣ ཚེས་ ༡༠ ཉིན་བོད་ཀྱི་རྒྱལ་ས་ལྷ་སར་བཙན་གནོན་བྱས་པར་རང་དབང་སྒེར་ ལངས་བྱ་དགོས་བྱུང་བ་དེ་ནི། བོད་ཀྱི་ལོ་རྒྱུས་སུ་དུས་ནམ་ཡང་བསུབ་ བ་ཏུ་མེད་པ་ཞིག་ཏུ་གྱུར་ཡོད།

བོད་ནང་ལོ་ངོ་སྟོང་ཕྲག་གི་ལོ་རྒྱུས་ཡོད་པའི་རྫོང་གཞིས་དང་། དགོན་པ་དང་ལྷ་ཁང་། བླ་བྲང་སོགས་བོད་མི་རིགས་ཀྱི་ཁྱད་ ཆོས་ལྡན་པ་ཡོད་དོ་ཅོག་གཏོར་བཤིག་བཏང་ནས་སྐུ་གསུང་ཐུགས་རྟེན་དང་། གསེར་དངུལ། ཟངས་ལྕགས་རིན་ཐང་ཅན གྱི་མཆོད་ཆས་རྣམས་དང་། དུད་ཁྱིམ་སོ་སོའི་རྟེན་ཆས་དང་། རྒྱན་ཆ་སོགས་རིན་ཐང་ལྡན་པའི་དངོས་རིགས་རྣམས་རྒྱ་ ནག་ཏུ་དབོར་འདྲེན་བྱས་པ་མ་ཟད། ཆོས་སྡེ་ཁག་གི་མཁན་བླ་སྤྲུལ་དང་། དགེ་བཤེས། ལས་སྣེ་དབུ་ཆོས་རྣམས་དང་། གཞུང་དང་རྫོང་གཞིས་ཁག་གི་མི་དྲག་ཚོར་སྐྱོན་བརྗོད་སྣ་ཚོགས་བྲིས་པའི་ཤོག་ཞྭ་གཡོགས་ཏེ་ཁྲོམ་སྐོར་དང་། མི་མང་འདུ་ འཛོམས་སར་ཁྲིད་པ། དེ་བཞིན་དགེ་ཕྲུག་གིས་དགེ་རྒན་དང་། གྲལ་རིམ་མཐོ་བའི་ཕྲུ་གུས་རང་གི་ཕ་མར་སྐྱོན་བརྗོད་ཀྱིས་ འཐབ་འཛིང་གཏོང་དུ་བཅུག་པ་མ་ཟད། བུད་མེད་ལ་བཙན་གཡེམ་སོགས་མི་སྤྱོད་ལ་འདས་པའི་སྡུག་སྦྱོང་བཏང་ཡོད་པ་དེ་ དག་ནི། བོད་མི་རིགས་ཀྱིས་བརྗེད་ཐབས་མེད་པའི་ལོ་རྒྱུས་ཀྱི་བྱུང་རིམ་ཞིག་ཆགས་ཡོད།
དེང་གི་ཆར་ཡང་། སྤྱི་ཚོགས་རིང་ལུགས་ཀྱི་བསམ་བློའི་སློབ་གསོ་དང་། རྒྱལ་གཅེས་རིང་ལུགས་ཀྱི་བསམ་བློའི་ སློབ་གསོ་ཟེར་བ་སོགས་བོད་མི་རིགས་རྩ་བརླག་གཏོང་རྒྱུའི་ལས་འགུལ་ཉིན་འཁྱོལ་ཡང་མཚན་མི་འཁྱོལ་བའི་དཀའ་ངལ་ འདྲ་མིན་ཟམ་མི་ཆད་པ་སྤེལ་དང་སྤེལ་མུས་ཡིན་ལ། དེ་དག་ལ་བོད་ནང་གི་བོད་མི་ཚོས་རང་མི་རིགས་ལ་བཅངས་པའི་ཆོད་ སེམས་རི་ལྟར་གཡོ་བ་མེད་པའི་ཐོག་སྙིང་སྟོབས་ཆེན་པོས་དངོས་ཤུགས་བརྒྱུད་གསུམ་ནས་གདོང་ལེན་གནང་བཞིན་ཡང་ རྒྱ་གཞུང་གིས་སྔ་ཕྱི་བར་གསུམ་དུ་ཆབ་སྲིད་ཀྱི་དགོས་མཁོ་ཁོ་ནའི་ཆེད་བོད་ནང་ཆབ་སྲིད་ཀྱི་གནོན་ཤུགས་དང་། རིག་ གཞུང་། སྤྱི་ཚོགས་དཔལ་འབྱོར་སོགས་ཀྱི་དབྱེ་འབྱེད་ཀྱིས་མཚོན་འགྲོ་བ་མི་ཡི་ཐོབ་ཐང་གི་གནས་སྟངས་ བཏང་དང་གཏོང་བཞིན་པ་ནི་རྒྱལ་སྤྱིའི་སྙན་ཐོ་མང་པོའི་ནང་ཐོན་ཡོད། སྟངས་ཇེ་ཞན་ཇེ་སྡུག་ཏུ་
དེ་བཞིན་བོད་ཀྱི་ཁོར་ཡུག་གཏོར་བཤིག་ཚད་མེད་བཏང་དང་གཏོང་བཞིན་པ་དང་། ལྷག་པར་དུ་བོད་རྒྱ་ཁོངས་སུ་ འགྲུབ་ཐབས་སུ་བོད་ཀྱི་རྒྱལ་རབས་ལོ་རྒྱུས་དང་། རིག་དངོས་གནའ་རྫས་སོགས་བེད་སྤྱོད་ལོག་པར་སྤྱད་དེ་འགྲེམས་སྟོན་ ཁང་དང་། དུས་དེབ། གསར་ཤོག བརྙན་འཕྲིན་སོགས་བརྒྱུད་ལམ་གང་ས་ནས་འཛམ་གླིང་ཡུལ་གྲུ་གང་སར་ཁུངས་ལུང་ བྲལ་བའི་ཁྱབ་སྤེལ་བཏང་དང་གཏོང་བཞིན་པ་དང་། ཆབས་ཅིག་ཏུ་བོད་ཀྱི་མཐུན་མོང་མ་ཡིན་པའི་སྲོལ་རྒྱུན་གྱི་རིག་གཞུང་ དང་། གོམས་སྲོལ། སྐད་ཡིག མི་རིགས་ཀྱི་ངོ་བོ་སོགས་རྩ་བརླག་ཏུ་གཏོང་རྒྱུའི་སྡུག་རྩུབ་ཅན་གྱི་སྲིད་བྱུས་མངོན་གསལ་ དོད་པོས་བོད་ནང་དུ་ལག་བསྟར་བྱེད་བཞིན་ཡོད། རྒྱ་གཞུང་ནས་"བོད་བརྒྱུད་ནང་བསྟན་ཀྲུང་གོ་ཅན་དུ་འགྱུར་བའི་ཁ་ ཕྱོགས་ཐོག་མཐའ་བར་གསུམ་དུ་རྒྱུན་འཁྱོངས་དང་། སྤྱི་ཚོགས་རིང་ལུགས་ཀྱི་ལྟེ་བའི་རིན་ཐང་ལྟ་བས་སྣེ་ཁྲིད་བྱེད་པ། ཀྲུང་ ཧྭ་ཕུལ་བྱུང་སྲོལ་རྒྱུན་རིག་གནས་ཀྱིས་ཤུགས་རྐྱེན་ཐེབས་པ་བཅས་བྱེད་དགོས།"ཞེས། བོད་བརྒྱུད་ནང་བསྟན་ཀྲུང་གོ་ ཅན་དུ་འགྱུར་རྒྱུའི་སྲིད་བྱུས་ལག་བསྟར་བྱེད་བཞིན་ཡོད། དེ་ནི་བོད་བརྒྱུད་ནང་བསྟན་གྱི་ལྟ་གྲུབ་དང་། བསླབ་གསུམ་གྱི་ ཉམས་ལེན། མདོ་སྔགས་ཀྱི་ཆོས་ཚུལ་དང་ཡོངས་སུ་འགལ་བ་ཞིག་ཡིན་པས། དེ་དག་འཕྲལ་དུ་འཚམས་འཇོག་བྱེད་དགོས་ པ་ཙམ་དུ་མ་ཟད། བོད་རིགས་གཙོ་བོར་གྱུར་པའི་ཕ་མེས་ནས་ནང་བསྟན་ལ་དད་སེམས་ཡོད་པའི་མི་མང་རྣམས་ལ་ཆོས་ དད་རང་དབང་སྤྲད་པའི་ཐོག ནང་པའི་གཞུང་ལུགས་ནས་བསྟན་པའི་བྱམས་བརྩེ་དང་འཚེ་བ་མེད་པའི་ལྟ་གྲུབ་གཞི་རྩར་ བཟུང་སྟེ། ཀྲུང་གོ་ཟེར་བའི་ཁ་ཕྱོགས་དེ་ནང་བསྟན་ལ་ཁ་ཕྱོགས་ཏེ་སྣེ་ཁྲིད་བྱས་ཚེ། སྤྱི་ཚོགས་འཆམ་མཐུན་དང་། བརྟན་ ལྷིང་། ཞི་མཐུན་མཉམ་གནས་ཡོང་རྒྱུ་ཙམ་དུ་མ་ཟད། འཛམ་གླིང་ཡུལ་གྲུ་ཀུན་ཏུ་ཞི་མཐུན་མཉམ་གནས་ཡོང་བར་ཤུགས་ རྐྱེན་ཐེབས་རྒྱུ་ཡིན་པ་སྨྲོས་མི་དགོས་པ་ཡིན་པས། རྒྱ་ནག་དབང་འཛིན་པ་ཚོས་ཁ་ཕྱོགས་མ་ནོར་བ་བྱ་དགོས་རྒྱུ་ཧ་ཅང་ གལ་ཆེན་པོ་ཡིན།
དེ་བཞིན་བོད་ཀྱི་མཐུན་མོང་མ་ཡིན་པའི་ཆོས་ཕྱོགས་ཀྱི་སྐྱེ་བ་སྔ་ཕྱིའི་སྤྲུལ་སྐུའི་ཡང་སྲིད་ངོས་འཛིན་གནང་ཕྱོགས་ ཐོག་རྒྱ་གཞུང་གིས་ཐེ་བྱུས་དང་ལག་ལེན་ལོག་པ་བྱས་དང་བྱེད་མུས་མ་ཟད། བོད་ཀྱི་ཆོས་ཕྱོགས་ཀྱི་སྲོལ་རྒྱུན་ལ་རྩིས་ མེད་རྡོག་རོལ་བཏང་སྟེ་སྐྱེ་བ་སྔ་ཕྱིའི་ཡང་སྲིད་དང་འབྲེལ་བའི་ཁྲིམས་གཞི་བཟོས་པ་རྣམས་སྔ་ཚིག་ཕྱི་འགལ་ཆགས་ཡོད། ལྷག་དོན་༸རྒྱལ་བའི་ཡང་སྲིད་ཕེབས་ཕྱོགས་ཐད་རྒྱ་གཞུང་གིས་དུས་ད་ལྟ་ནས་གྲ་སྒྲིག་བྱེད་མུས་དང་། དེ་འབྲེལ་བརྒྱུད་ ་འཛམ་གླིང་ཐོག་༸གོང་ས་མཆོག་གི་སྐུའི་ཡང་སྲིད་རྒྱ་ནག་ནས་ངོས་འཛིན་བྱེད་དགོས་པ་སོགས་དགོས་ ལམ་བེད་སྤྱད་དེ་འཛམ་ མཁོའི་དབང་གིས་ཆབ་སྲིད་ཀྱི་བྱེད་དབང་ཞིག་ལས་ཁུངས་ལུང་གང་ཡང་མེད་པའི་སྨྱོ་གཏམ་བསྒྲགས་དང་བསྒྲགས་མུས་ རེད། བོད་ཀྱི་སྲོལ་རྒྱུན་བླ་མའི་ཡང་སྲིད་ཕེབས་ཕྱོགས་ནི་བླ་མ་རང་ཉིད་གང་དེར་ཡོད་པ་ལས་གཞན་སུ་ཞིག་གིས་བཙན་ ཤེད་བྱ་ཡུལ་མེད། ལྷག་པར་དུ་ྋགོང་ས་མཆོག་གི་སྐུའི་ཡང་སྲིད་ཐད་མགོན་པོ་གང་ཉིད་ནས་སྔ་རྗེས་བཀའ་སློབ་གསལ་པོ་ ཕེབས་ཡོད་པའི་ནང་དོན་ལ་བོད་ཕྱི་ནང་གཉིས་དང་། རྒྱལ་སྤྱིའི་ཁྱོན་ནས་དེ་ལྟར་བརྩི་འཇོག་ཞུ་ངེས་ཆགས་ཡོད།
སྔ་ལོ་རྒྱ་གཞུང་གིས་ལོ་ལྔ་ཚན་བཅུ་བཞི་པའི་སྐད་ཡིག་ཚད་ལྡན་འཛུགས་སྐྲུན་འཆར་འགོད་ཟེར་བའི་ཁོངས་སུ། སློབ་གྲྭ་ཆུང་འབྲིང་གི་སློབ་མའི་སྤྱི་སྐད་ཆུ་ཚད་ཀྱི་ཚད་རིམ་རྒྱུགས་གཞིའི་ཚད་གཞི་དང་རྒྱུགས་གཞིའི་རྩ་འཛིན་ཟེར་བ་ཞིག་ གཏན་འབེབ་བྱས་ཡོད། དེ་ནི་ཕྱི་ལོ་ ༢༠༠༩ ལོར་རྒྱ་ནག་ཤེས་རིག་ལྷན་ཁང་ཟེར་བ་དང་སྐད་ཡིག་འཛིན་སྐྱོང་ལྷན་ཁང་ཟེར་ བ་གཉིས་ཀྱིས་གཏན་འབེབས་བྱས་པའི་སློབ་ཆུང་དང་དམའ་རིམ་སློབ་མའི་བསླབ་གཞི་ཡོད་ཚད་རྒྱ་ཡིག་ཐོག་ཏུ་བསྒྱུར་བའི་ སྲིད་ཇུས་ཀྱི་ཆ་ཤས་ཤིག་ཡིན་ལ། ཕྱི་ལོ་ ༢༠༣༥ བར་དུ་རྒྱའི་དབང་འོག་ཏུ་ཡོད་པའི་མི་རིགས་གཞན་ཁག་གི་སྐད་ཡིག་རྩ་ མེད་བཟོས་ཏེ་སྤྱི་སྐད་ཅེས་རྒྱ་སྐད་ཅན་དུ་བསྒྱུར་རྒྱུའི་སྲིད་བྱུས་འདི་ཉིད་མི་རིགས་ས་གནས་རང་སྐྱོང་གི་ཐོབ་ཐང་ཟེར་བའི་ ནང་དོན་དང་དངོས་སུ་འགལ་བའི་བྱེད་ཕྱོགས་ཤིག་ཡིན། སྲིད་བྱུས་དེ་ད་ཆ་བོད་ནང་དུ་ལག་བསྟར་ཤུགས་ཆེན་པོས་སློབ་ ཕྲུག་དང་ཕ་མའི་ངོས་ནས་རྒྱའི་སྐད་ཡིག་ལ་ཤུགས་སྣོན་མ་རྒྱག་ཀ་མེད་ཀྱི་གནས་སྟངས་བཟོ་བཞིན་ཡོད།
དེང་སྐབས་བོད་ནང་རྒྱ་གཞུང་གིས་སྟངས་འཛིན་འོག་བཅའ་སྡོད་སློབ་གྲྭ་ཟེར་བ་ཕོན་ཆེ་ཆེད་འཛུགས་བྱས་དང་ བྱེད་བཞིན་པ་དང་། ཕ་མ་དང་ཕྲུ་གུར་འདོད་པ་མེད་བཞིན་དུ་བཅའ་སྡོད་སློབ་གྲྭའི་ནང་མ་གཏང་ཀ་མེད་ཀྱི་འཇིགས་སྐུལ་དང་ གནོན་ཤུགས་སྤྲོད་བཞིན་ཡོད། སློབ་ཕྲུག་གི་གྲངས་འབོར་ཡང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱིས་དམིགས་བསལ་ཞིབ་འཇུག་ མཁན་བཞིས་བྱས་པའི་(Four UN Special Rapporteurs) སྙན་ཐོ་ལྟར་ན་ཕྲུ་གུ་ས་ཡ་གཅིག་མན་ཙམ་ཡིན་པ་ གསལ་ཡོད། དེའི་ཁོངས་སུ་བྱིས་པ་ལོ་ན་ ༤ དང་ ༥ ཙམ་ཟིན་པ་ཡང་འཇུག་བཞིན་ཡོད། སློབ་མ་དེ་དག་བོད་མི་རིགས་ཀྱི་ ངོ་བོ་དང་འདུ་ཤེས་ཀྱི་རྣམ་པ་ཙམ་ཡང་ ཡང་ཤེས་ཐབས་མེད་པའི་ཁོར་ཡུག་ནང་བཅུག་སྟེ། རྒྱ་ནག་དམར་གཞུང་ལ་དགའ་མོས་ བྱེད་མཁན་གྱི་མི་རབས་ཤིག་བཟོ་ཐབས་བྱེད་བཞིན་ཡོད། སློབ་གྲྭ་དེ་དག་གི་ནང་མཐུན་རྐྱེན་ཧ་ཅང་གི་ཞན་པོ་ཡོད་པ་མ་ ཟད། སློབ་ཕྲུག་རྣམས་རང་གི་ཕ་མ་དང་སྤྱི་ཚོགས་ཁ་བྲལ་བའི་རྐྱེན་གྱིས་གྲོགས་མེད་ཁྱེར་རྐྱང་དུ་ལུས་པའི་དཀའ་ངལ་ ཤུགས་ཆེ་འཕྲད་བཞིན་པའི་ཁར། བྱིས་པའི་འཚར་ལོངས་དང་སེམས་ཁམས། འཕྲོད་བསྟེན་སོགས་ལ་གནོད་པའི་རྡུང་རྡེག་ དང་། བཙན་གཡེམ། དབྱེ་འབྱེད་སོགས་བྱེད་པའི་གནས་ཚུལ་སྐྱི་གཡའ་བ་མང་དག་ཅིག་ཐོན་དང་ཐོན་མུས་ཡིན་པས། བཅའ་སྡོད་སློབ་གྲྭ་ཞེས་བོད་མི་རིགས་ཀྱི་ངོ་བོ་དང་རིག་གཞུང་རྩ་མེད་བཟོ་བའི་སྲིད་བྱུས་ལག་བསྟར་བྱེད་བཞིན་པ་ཐེར་ འདོན་དང་འབྲེལ། དེ་རིང་གི་གོ་སྐབས་འདི་བརྒྱུད་ནས་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱིས་གཙོས་རྒྱལ་སྤྱིའི་རྒྱལ་ཁབ་ཀྱི་དབུ་ ཁྲིད་རྣམས་ཀྱིས་ངེས་པར་དུ་རྒྱ་གཞུང་ལ་བཅའ་སྡོད་སློབ་གྲྭའི་ལས་གཞི་སྤེལ་མཚམས་འཇོག་དགོས་པའི་སྐུལ་མ་དང་ གནོན་ཤུགས་སྤྲོད་དགོས་པའི་འབོད་བསྐུལ་ཞུ་རྒྱུ་ཡིན།
དེ་བཞིན་རྒྱ་གཞུང་གིས་བོད་རང་ལྗོངས་ཟེར་བའི་ནང་བོད་མིའི་རིགས་རྫས་ DNA བསྡུ་ལེན་གྱི་ལས་གཞི་འདི་ ཉིད་ཉམས་ཞིབ་པ་འགས་བཤད་པ་ལྟར་ན་ཕྱི་ལོ་ ༢༠༡༣ ཙམ་ནས་མངོན་གསལ་མེད་པའི་ཐོག་ནས་འགོ་འཛུགས་བྱས་ ཡོད་པ་དང་། དེང་སང་བོད་རིགས་བྱིས་པ་ལོ་ན་ལྔ་ཡན་གྱི་རིགས་རྫས་བསྡུ་ལེན་བྱེད་བཞིན་ཡོད་པ་ཨ་རིར་རྟེན་གཞི་བྱས་ པའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ལྟ་ཞིབ་ (Human Rights Watch) ཚོགས་པའི་སྙན་ཐོར་བཀོད་ཡོད། མདོར་ན་རྒྱ་གཞུང་ གིས་བོད་ས་བཙན་བཟུང་བྱས་པ་ནས་ད་བར་མི་སེར་སྤེལ་བའི་སྲིད་བྱུས་འོག་བོད་མིའི་མངོན་འདོད་ལ་ཁྱད་གསོད་ཀྱིས་ དྲག་གནོན་གྱི་སྲིད་བྱུས་ཁོ་ན་ལག་བསྟར་བྱས་དང་བྱེད་བཞིན་པར་བརྟེན། ཕྱི་ལོ་ ༡༩༥༩ དང༌། ཕྱི་ལོ་ ༡༩༨༧།༡༩༨༨།༡༩༨༩ ཞི་རྒོལ་རིམ་པར་བྱུང་བ་ཁག་དང་། ཕྱི་ལོ་ ༢༠༠༨ ལོར་ས་བྱི་ཞི་རྒོལ། ༢༠༠༩ ལོ་ནས་རང་ལུས་ མེར་བསྲེགས་སོགས་ཀྱི་ཞི་རྒོལ་གནང་བ་དག་གིས་མཚོན་སྔ་ཕྱི་བར་གསུམ་དུ་རྒྱ་གཞུང་གི་ནོར་འཁྲུལ་ཅན་གྱི་སྲིད་བྱུག ལ་བོད་མི་ཚོས་ཞི་རྒོལ་གནང་བ་མཚམས་ཆད་མྱོང་མེད་ལ། དེ་ནི་བོད་དོན་བདེན་མཐའ་མ་གསལ་བར་དུ་མཚམས་ཆད་མི་ སྲིད་པའི་གནས་སྟངས་ཤིག་ཆགས་ཡོད།
ཕྱི་ལོ་ ༢༠༡༩ ལོའི་ལོ་མཇུག་ཏུ་རྒྱ་ནག་ཝུའུ་ཧན་ནས་མཆེད་པའི་ཏོག་དབྱིབས་གཉན་རིམས་འཛམ་གླིང་ཡུལ་ ལུང་གང་སར་ཁྱབ་གདལ་ཕྱིན་ནས་ཕྱི་ལོ་ ༢༠༢༣ ཟླ་ ༢ ཚེས་ ༢༥ བར་མི་གྲངས་ས་ཡ་བདུན་བརྒྱ་ལྔ་བཅུ་ང་བརྒྱད་ལྷག་ ཙམ་ནད་མནར་དང་། ས་ཡ་བདུན་མ་ཟིན་ཙམ་རྐྱེན་འདས་སུ་ཕྱིན་པ་འཛམ་གླིང་འཕྲོད་བསྟེན་ལྷན་ཚོགས་ཀྱི་སྙན་ཐོར་ བཀོད་ཡོད། སྔ་ལོ་རྒྱ་གཞུང་གིས་ནད་ཡམས་ཀླད་ཀོར་ཟེར་བའི་སྲིད་བྱུས་འོག་འགྲོ་འོང་བཀག་བསྡོམས་དམ་དྲག་ཤུགས་ ཆེ་བྱས་ཡོད་ཀྱང་ཟླ་ ༡༡ པའི་ཟླ་མཇུག་ཙམ་ནས་རྒྱ་ནག་གི་ས་ཁུལ་གང་ས་ནས་སྲིད་བྱུས་དེར་ངོ་རྒོལ་ཤུགས་ཆེ་བྱས་ སྟབས་ནད་ཡམས་ཀླད་ཀོར་ཟེར་བའི་སྲིད་བྱུས་ལ་ཕམ་ཉེས་བྱུང་ནས་འགྲོ་འོང་བྱེད་དུ་མི་འཇུག་ཀ་མེད་ཆགས་སྟབས་ནད་ ཡམས་ཁྱབ་གདལ་ཆེན་པོ་ཕྱིན་པ་དང་། སྐབས་དེའི་བོད་ནང་གི་གྲངས་ཐོ་ངེས་ཅན་ཤེས་རྟོགས་དཀའ་ནའང་བོད་ནང་སྨན་ བཅོས་ཀྱི་མཐུན་རྐྱེན་འདང་ངེས་མེད་པ་སོགས་ཀྱིས་ནད་མནར་དང་རྐྱེན་འདས་སུ་ཕྱིན་པའི་བརྙན་པར་བོད་ཀྱི་ལོ་རྒྱུས་སུ་ བྱུང་མྱོང་མེད་པ་སྤྱི་ཚོགས་དྲ་ལམ་ཐོག་ཐོན་ཡོད། དེ་རིང་གི་གོ་སྐབས་འདི་བརྒྱུད་ནས་ཁོང་རྣམས་ལ་གདུང་སེམས་མཉམ་ བསྐྱེད་མཚོན་ཆེད་ཡིད་སྐྱོ་བློ་ཕམ་དང་བཅས་འདས་པོ་རྣམས་ལ་བསྔོ་སྨོན་དང་། ཤུལ་ལུས་རྣམས་ལ་ཐུགས་གསོ་ཞུ་རྒྱུ་ ཡིན།
རྒྱ་གཞུང་གིས་བོད་མི་རིགས་རྩ་མེད་བཟོ་བའི་སྲིད་བྱུས་དང་མི་རིགས་དབྱེ་འབྱེད་ཀྱི་གཉའ་གནོན་ལ་གཞིས་ བཞུགས་བོད་མི་རྣམས་ཀྱིས་མགོ་བོ་ནམ་ཡང་མི་སྒུར་བར་ཞུམ་པ་མེད་པའི་སྙིང་སྟོབས་ཆེར་བསྐྱེད་ཐོག་ནས་ཞི་བའི་ལས་ འགུལ་སྤེལ་ པ་བ་རྣམས་ཆུད་ཟོས་སུ་མི་འགྲོ་བ་དང་། རྒྱལ་སྤྱིའི་ཁྱོན་ཡོངས་ཀྱི་རྒྱལ་ཁབ་མང་པོ་ཞིག་གིས་བོད་དོན་ཐོག་རྒྱབ་ སྐྱོར་ཡོད་པའི་ཁྲིམས་གཞི་དང་གྲོས་ཆོད་རིམ་པར་བཞག་པ་དང་། བསྒྲགས་གཏམ། སེམས་འཚབ་ཡོད་པའི་གཏམ་བཤད་ གནང་བ་སོགས་ཀྱིས་མཚོན་རྒྱ་ནག་གཞུང་ལ་དངོས་ཤུགས་བརྒྱུད་གསུམ་ནས་བོད་ནང་གི་དྲག་གནོན་གྱི་སྲིད་བྱུས་ མཚམས་འཇོག་དགོས་པའི་རེ་འདུན་དང་། གཉིས་སྨན་དབུ་མའི་ལམ་གྱི་ཐོག་ནས་བོད་རྒྱའི་དཀའ་རྙོག་སེལ་རྒྱུར་འབྲེལ་ མོལ་ཡོང་ཆེད་རྒྱ་གཞུང་གིས་ྋགོང་ས་མཆོག་གི་སྐུ་ཚབ་དང་ལྷན་གྲོས་མོལ་དགོས་པའི་འབོད་སྐུལ་ཤུགས་ཆེ་གནང་ཡོད་ ལ། ད་ལྟའི་བོད་ནང་གི་འགྲོ་བ་མིའི་ཐོབ་ཐང་གི་གནས་སྟངས་ཡར་རྒྱས་ཡོང་བ་དང་། བོད་དོན་གྱི་དཀའ་རྙོག་སེལ་ཐབས་སུ་ གྲོས་མོལ་ལ་ང་ཚོའི་ངོས་ནས་ཆོད་སེམས་ལ་འགྱུར་བ་གང་ཡང་མེད་པར་བརྟེན། རྒྱ་གཞུང་གི་དབང་འཛིན་པ་ཚོས་གོ་ སྐབས་དམ་འཛིན་གྱིས་བོད་རྒྱ་འབྲེལ་མོལ་མྱུར་དུ་མཐུད་དགོས་པའི་ནན་སྐུལ་བྱེད་ཀྱི་ཡོད།
འདས་ལོའི་ནང་ཨ་རིའི་རྒྱལ་ས་ཝ་ཤིན་ཊོན་དུ་གྲོས་ཚོགས་བོད་དོན་རྒྱབ་སྐྱོར་ཚོགས་ཆེན་ཐེངས་ ༨ པ་བསྐོངས་ ཚོགས་གྲུབ་རྗེས་ལྷོ་ཨ་མི་རི་ཀའི་ནང་མེཀ་སི་ཀོ་དང་། ཉེ་ཆར་སི་པན་གྲོས་ཚོགས་གོང་མའི་ནང་བོད་དོན་རྒྱབ་སྐྱོར་ཚོགས་ པ་གསར་འཛུགས་བྱུང་ཐུབ་པ་ནི་བོད་དོན་འཐབ་རྩོད་ཐད་རྒྱལ་ཁ་ཆེན་པོ་ཡིན་ལ་འབྲེལ་ཡོད་འགན་འཁུར་གནང་མཁན་ ཡོངས་ལ་བསྔགས་བརྗོད་ཞུ་རྒྱུ་དང༌། དེ་རིང་གི་མཛད་སྒོ་འདིར་ཡུ་རོབ་གྲོས་ཚོགས་ (Member of European Parliament) དང་། མེག་སི་ཀོ། (Mexico) ལི་ཐི་ནི་ཡ་ (Lithuania) བཅས་ནས་གྲོས་ཚོགས་ཀྱི་འཐུས་མི་ཆེད་ ཕེབས་ཀྱིས་བདེན་པ་དང་འཚེ་བ་མེད་པའི་བོད་དོན་ལ་རྒྱབ་སྐྱོར་གནང་བ་ནི་ཧ་ཅང་གི་གལ་ཆེན་པོ་ཞིག་ཆགས་ཡོད་པས་ ཁོང་རྣམས་ལ་དམིགས་བསལ་གྱིས་ཐུགས་རྗེ་ཆེ་ཞུ་རྒྱུ་ཡིན།
བོད་མི་མང་སྤྱི་འཐུས་ལྷན་ཚོགས་ནས་སྤྱི་འཐུས་རྣམ་པ་ཆ་བགོས་ཀྱིས་བོད་དོན་ཞུ་སྐུལ་གྱི་ལས་འགུལ་སྤེལ་དང་ སྤེལ་མུས་ཡིན་ལ་འཆར་ལོའི་ནང་གཞི་རྒྱ་ཆེ་ཙམ་གྱི་ལས་འགུལ་སྤེལ་རྒྱུའི་འཆར་གཞི་ཡོད། བོད་དོན་བདེན་མཐའ་གསལ་ བར་གལ་འགངས་ཆེ་ཤོས་ནི་བོད་པ་རང་ཉིད་ཀྱིས་འགན་ཁུར་དགོས་པས་གཞིས་བཞུགས་བོད་མི་རྣམས་ཀྱི་སྙིང་སྟོབས་ དང་། ལྷག་བསམ། ཆོད་སེམས་བཅས་རི་ལྟར་ཡོད་པ་དང་། བཙན་བྱོལ་བོད་མི་རྣམས་ཀྱིས་སྤྱི་ནོར་༸གོང་ས་ྋསྐྱབས་ མགོན་ཆེན་པོ་མཆོག་གི་ཐུགས་རྗེ་བཀའ་དྲིན་དང་། རང་དབང་མང་གཙོའི་ཁེ་དབང་ལོངས་སུ་སྤྱད་དེ་བོད་ཀྱི་བདེན་པའི་ འཐབ་རྩོད་ཐད་ངོ་གནོང་མི་དགོས་པ་ཞིག་བྱ་དགོས་རྒྱུ་ཧ་ཅང་གི་གལ་ཆེན་ཡིན།
དེ་རིང་གི་གོ་སྐབས་འདི་དང་བསྟུན་བཙན་བྱོལ་བོད་མི་རྣམས་ལ་མཐུན་རྐྱེན་སྦྱོར་སྤྲོད་གནང་མཁན་བཀའ་དྲིན་ཆེ་ བའི་རྒྱ་གར་གཞུང་མང་གཉིས་ཀྱིས་གཙོས་བོད་དོན་ལ་ཐུགས་ཁུར་རྒྱབ་སྐྱོར་གནང་མཁན་གྱི་རྒྱལ་ཁབ་དང་། ཚོགས་པ། མི་སྒེར་བཅས་ལ་གཞིས་བྱེས་བོད་མི་ཡོངས་ཀྱི་ཚབ་ཞུས་ཏེ་བཀའ་དྲིན་བླ་མེད་དུ་ཆེ་ཞུ་རྒྱུ་ཡིན།
མཐའ་དོན། སྤྱི་ནོར་ྋགོང་ས་ྋསྐྱབས་མགོན་ཆེན་པོ་མཆོག་བསྟན་འགྲོའི་མགོན་དཔུང་དམ་པར་༧ ༧ཞབས་པད་ བསྐལ་བརྒྱའི་བར་དུ་བརྟན་ཅིང་། རླབས་ཆེན་ཐུགས་ཀྱི་དགོངས་བཞེད་རྣམས་ལྷུན་གྱིས་འགྲུབ་པ་དང་། བོད་དོན་བདེན་ མཐའ་མྱུར་དུ་གསལ་ཏེ་གཞིས་བྱེས་མཉམ་འཛོམས་ཡོང་བར་སྨོན་འདུན་དང་བཅས། བོད་མི་མང་སྤྱི་འཐུས་ལྷན་ཚོགས་ ནས། ཕྱི་ལོ་ ༢༠༢༣ ཟླ་ ༣ ཚེས་ ༡༠ ལ།།
